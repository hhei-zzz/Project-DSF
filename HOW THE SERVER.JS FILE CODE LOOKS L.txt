require("dotenv").config();
const express = require("express");
const cors = require("cors");
const mysql = require("mysql2/promise");
const fs = require("fs");

const app = express();
app.use(cors());
app.use(express.json());

const DB_HOST = process.env.DB_HOST;
const DB_USER = process.env.DB_USER;
const DB_PASS = process.env.DB_PASS;
const DB_NAME = process.env.DB_NAME;

let sslConfig = undefined;
if (fs.existsSync("./global-bundle.pem")) {
  sslConfig = { ca: fs.readFileSync("./global-bundle.pem") };
}

const pool = mysql.createPool({
  host: DB_HOST,
  user: DB_USER,
  password: DB_PASS,
  database: DB_NAME,
  port: 3306,
  ssl: sslConfig,
});

async function getCheckPasses(playerId) {
  const gameId = await getPlayerGameId(playerId);

  const [moves] = await pool.query(
    "SELECT player_id, move_type FROM move WHERE game_id = ? ORDER BY id DESC",
    [gameId]
  );

  let passCounter = 0;

  for (let i = 0; i < moves.length; i++) {
    const move = moves[i];

    if (move.move_type === "CLEAR") {
      break;
    }

    if (move.move_type === "PASS") {
      passCounter++;
      continue;
    }

    if (move.move_type === "PLAY") {
      if (passCounter === 3) {
        return {
          board_cleared: true,
          lastPlayPlayerId: move.player_id
        };
      }

      return {
        board_cleared: false,
        lastPlayPlayerId: null
      };
    }
  }

  return {
    board_cleared: false,
    lastPlayPlayerId: null
  };
}

// kollar vems tur det är just nu, om det är playerId 2 som ska spela så kan inte playerId spela
async function getCurrentTurnPlayerId(playerId) {
  // få game_id från denna spelaren
  const [playerRows] = await pool.query(
    "SELECT game_id FROM player WHERE id = ? LIMIT 1",
    [playerId]
  );

  const gameId = playerRows[0].game_id;

  // få vems tur det är från "game" tabellen 
  const [gameRows] = await pool.query(
    "SELECT current_turn_player_id FROM game WHERE id = ? LIMIT 1",
    [gameId]
  );

  return gameRows[0].current_turn_player_id;
}

async function getNextPlayerId(playerId){
  const [playerRows] = await pool.query(
    "SELECT game_id, seat FROM player WHERE id = ? LIMIT 1",
    [playerId]
  );

  const gameId = playerRows[0].game_id;
  const currentSeat = playerRows[0].seat;
  
  const [allPlayers] = await pool.query (
    "SELECT id, seat FROM player WHERE game_id = ? ORDER BY seat",
    [gameId]
  );

  const currentIndex = allPlayers.findIndex(player => player.seat === currentSeat);
  const nextIndex = (currentIndex + 1) % allPlayers.length;

  return allPlayers[nextIndex].id;

}

async function checkRoundWinnerAndReset(playerId) {
  const [playerRows] = await pool.query(
    "SELECT game_id FROM player WHERE id = ? LIMIT 1",
    [playerId]
  );

  if (playerRows.length === 0) {
    throw new Error("Player not found");
  }

  const gameId = playerRows[0].game_id;

  const [cardRows] = await pool.query(
    `SELECT COUNT(*) AS count
     FROM player_card
     WHERE game_id = ? AND player_id = ? AND in_hand = 1`,
    [gameId, playerId]
  );

  const cardsLeft = cardRows[0].count;

  if (cardsLeft > 0) {
    return { winner: false };
  }

  const winnerId = playerId;

  await pool.query(
    "INSERT INTO move (game_id, player_id, move_type) VALUES (?, ?, 'CLEAR')",
    [gameId, winnerId]
  );

  await pool.query(
    "DELETE FROM player_card WHERE game_id = ?",
    [gameId]
  );

  await pool.query(
    "UPDATE player SET ready = 0 WHERE game_id = ?",
    [gameId]
  );

  await pool.query(
    "UPDATE game SET current_turn_player_id = ?, status = 'WAITING' WHERE id = ?",
    [winnerId, gameId]
  );

  return { winner: true, winnerId };
}

async function getPlayerGameId(playerId) {
  const [rows] = await pool.query(
    "SELECT game_id FROM player WHERE id = ? LIMIT 1",
    [playerId]
  );

  if (rows.length === 0) {
    throw new Error("Player not found");
  }

  return rows[0].game_id;
}


app.post("/join", async (req, res) => {
  try {
    const { name } = req.body;

    if (!name || !name.trim()) {
      return res.status(400).json({ error: "Name is required" });
    }

    const trimmedName = name.trim();

    // hitta ett game med status "WAITING" med mindre än 4 spelare
    const [waitingGames] = await pool.query(`
      SELECT g.id
      FROM game g
      LEFT JOIN player p ON g.id = p.game_id
      WHERE g.status = 'WAITING'
      GROUP BY g.id
      HAVING COUNT(p.id) < 4
      ORDER BY g.id ASC
      LIMIT 1
    `);

    let gameId;

    if (waitingGames.length > 0) {
      gameId = waitingGames[0].id;
    } else {
      // skapar en ny game session
      const [gameResult] = await pool.query(
        "INSERT INTO game (status) VALUES ('WAITING')"
      );
      gameId = gameResult.insertId;
    }

    // kollar om spelaren redan finns/joinat gamelobbyn som är "WAITING", om det gör, returnering den som finns. Inga dubletter skapas
    const [existing] = await pool.query(
      "SELECT id, name, seat, game_id FROM player WHERE game_id = ? AND name = ? LIMIT 1",
      [gameId, trimmedName]
    );

    if (existing.length > 0) {
      return res.json({
        player_id: existing[0].id,
        name: existing[0].name,
        seat: existing[0].seat,
        game_id: existing[0].game_id
      });
    }

    // placerar spelaren i nästa lediga stol
    const [seatRows] = await pool.query(
      "SELECT seat FROM player WHERE game_id = ? ORDER BY seat",
      [gameId]
    );

    const usedSeats = seatRows.map(r => r.seat);
    let nextSeat = 1;
    while (usedSeats.includes(nextSeat) && nextSeat <= 4) {
      nextSeat++;
    }

    if (nextSeat > 4) {
      return res.status(400).json({ error: "Game is full" });
    }

    const joinToken = Math.random().toString(36).slice(2, 12);

    const [result] = await pool.query(
      "INSERT INTO player (game_id, name, seat, join_token) VALUES (?, ?, ?, ?)",
      [gameId, trimmedName, nextSeat, joinToken]
    );

    res.json({
      player_id: result.insertId,
      name: trimmedName,
      seat: nextSeat,
      game_id: gameId
    });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Join failed" });
  }
});

app.post("/ready", async (req, res) => {
  try {
    const { player_id } = req.body;

    if (!player_id) {
      return res.status(400).json({ error: "player_id is required" });
    }

    const gameId = await getPlayerGameId(player_id);

    await pool.query(
      "UPDATE player SET ready = 1 WHERE id = ?",
      [player_id]
    );

    const [players] = await pool.query(
      "SELECT id, seat, ready FROM player WHERE game_id = ? ORDER BY seat",
      [gameId]
    );

    if (players.length < 4) {
      return res.json({
        status: "waiting",
        message: "Waiting for 4 players to join"
      });
    }

    const allReady = players.every(p => p.ready === 1);

    if (!allReady) {
      return res.json({
        status: "waiting",
        message: "Waiting for other players to click ready"
      });
    }

    const [existingCards] = await pool.query(
      "SELECT id FROM player_card WHERE game_id = ? LIMIT 1",
      [gameId]
    );

    // dela endast ut korten om det inte finns några kort i hand
    if (existingCards.length === 0) {
      const [cards] = await pool.query(
        "SELECT id FROM card ORDER BY RAND()"
      );

      const values = [];
      for (let i = 0; i < 52; i++) {
        const playerIndex = Math.floor(i / 13);
        values.push([gameId, players[playerIndex].id, cards[i].id, 1]);
      }

      await pool.query(
        "INSERT INTO player_card (game_id, player_id, card_id, in_hand) VALUES ?",
        [values]
      );
      
      //logik för första rundan endast, om moves för nuvarande game är tom, leta efter ruter 3 som börjar
      const [moves] = await pool.query(
        "SELECT id FROM move WHERE game_id = ? LIMIT 1",
        [gameId]
      );

      if (moves.length === 0) {
        const [whoStarts] = await pool.query(
          "SELECT player_id FROM player_card WHERE game_id = ? AND card_id = 2 LIMIT 1",
          [gameId]
        );

        if (whoStarts.length === 0) {
          return res.status(500).json({ error: "Could not find starter" });
        }

        const starterId = whoStarts[0].player_id;

        await pool.query(
          "UPDATE game SET current_turn_player_id = ? WHERE id = ?",
          [starterId, gameId]
        );
      }
      // blir en else sats här, vinnarna lagras redan i current_turn_player_id så rundorna efter startas utifrån det
    }

    await pool.query(
      "UPDATE game SET status = 'ACTIVE' WHERE id = ?",
      [gameId]
    );

    res.json({
      status: "all_ready",
      board_cleared: true,
      message: "All players ready, cards dealt",
      game_id: gameId
    });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Ready failed" });
  }
});

app.post("/play", async (req, res) => {
  try {
    const { player_id, cards } = req.body;

    if (!player_id || !Array.isArray(cards) || cards.length === 0) {
      return res.status(400).json({ error: "player_id and cards are required" });
    }

    const gameId = await getPlayerGameId(player_id);
    const currentTurnPlayerId = await getCurrentTurnPlayerId(player_id);

    if (player_id !== currentTurnPlayerId) {
      return res.status(400).json({ error: "Not your turn!" });
    }

    for (const card of cards) {
      await pool.query(
        `UPDATE player_card
         SET in_hand = 0
         WHERE game_id = ? AND player_id = ? AND card_id = ?`,
        [gameId, player_id, card.card_id]
      );
    }

    await pool.query(
      "INSERT INTO move (game_id, player_id, move_type, cards_json) VALUES (?, ?, 'PLAY', ?)",
      [gameId, player_id, JSON.stringify(cards)]
    );

    const result = await checkRoundWinnerAndReset(player_id);

    if (result.winner) {
      const [winnerRow] = await pool.query(
        "SELECT name FROM player WHERE id = ? LIMIT 1",
        [result.winnerId]
      );

      return res.json({
        status: "round_won",
        winner: winnerRow[0].name,
        game_id: gameId
      });
    }

    const nextPlayerId = await getNextPlayerId(player_id);

    await pool.query(
      "UPDATE game SET current_turn_player_id = ? WHERE id = ?",
      [nextPlayerId, gameId]
    );

    res.json({ status: "ok", game_id: gameId });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Play failed" });
  }
});

app.post("/pass", async (req, res) => {
  try {
    const { player_id } = req.body;

    if (!player_id) {
      return res.status(400).json({ error: "player_id is required" });
    }

    const gameId = await getPlayerGameId(player_id);
    const currentTurnPlayerId = await getCurrentTurnPlayerId(player_id);

    if (player_id !== currentTurnPlayerId) {
      return res.status(400).json({ error: "Not your turn" });
    }

    await pool.query(
      "INSERT INTO move (game_id, player_id, move_type) VALUES (?, ?, 'PASS')",
      [gameId, player_id]
    );

    const result = await getCheckPasses(player_id);

    if (result.board_cleared) {
      await pool.query(
        "UPDATE game SET current_turn_player_id = ? WHERE id = ?",
        [result.lastPlayPlayerId, gameId]
      );

      await pool.query(
        "INSERT INTO move (game_id, player_id, move_type) VALUES (?, ?, 'CLEAR')",
        [gameId, result.lastPlayPlayerId]
      );

      return res.json({
        status: "board_cleared",
        board_cleared: true,
        next_player: result.lastPlayPlayerId,
        game_id: gameId
      });
    }

    const nextPlayerId = await getNextPlayerId(player_id);

    await pool.query(
      "UPDATE game SET current_turn_player_id = ? WHERE id = ?",
      [nextPlayerId, gameId]
    );

    res.json({
      status: "ok",
      board_cleared: false,
      game_id: gameId
    });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Pass failed" });
  }
});

app.get("/history/:playerId", async (req, res) => {
  try {
    const playerId = Number(req.params.playerId);
    const gameId = await getPlayerGameId(playerId);

    const [rows] = await pool.query(`
      SELECT 
        m.id,
        m.player_id,
        p.name AS player_name,
        m.move_type,
        m.cards_json,
        m.created_at
      FROM move m
      JOIN player p ON m.player_id = p.id
      WHERE m.game_id = ?
      ORDER BY m.id DESC
    `, [gameId]);

    const currentBoardMoves = [];

    for (const row of rows) {
      if (row.move_type === "CLEAR") {
        break;
      }

      currentBoardMoves.push(row);
    }

    currentBoardMoves.reverse();

    res.json(currentBoardMoves);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Could not load history" });
  }
});

    const [rows] = await pool.query(`
      SELECT 
        g.current_turn_player_id,
        p.name AS current_turn_player_name,

  } catch (err) {
    console.error(err);
